```javascript
// 1. Check if user has access
const accessibleFrameIds = await paymentDB.getUserAccessibleFrames(userId);
const accessibleSet = new Set(accessibleFrameIds.map((id) => String(id)));

// 2. For each frame, check premium status
frames.map((frame) => {
  const isPremium = !!frame.is_premium;
  const canAccess = !isPremium || accessibleSet.has(String(frame.id));

  // 3. Redact sensitive data if no access
  return {
    ...frame,
    slots: canAccess ? frame.slots : [], // Empty if locked
    layout: canAccess
      ? frame.layout
      : {
          ...frame.layout,
          elements: [], // No elements if locked
        },
    isPremium: isPremium,
    isLocked: isPremium && !canAccess, // Lock flag
  };
});
```
